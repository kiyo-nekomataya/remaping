/*IMPORT MODULES inHTML*/
<script type='module' src = ./uaf-modules/index-nas.js.js >
/*IMPORT MODULES inCode*/
if (typeof require == 'function'){
	var nas = require("./uaf-modules/index-nas.js");
} else if (typeof navigator != 'undefined'){
	var nas = import("./uaf-modules/index-nas.js");
}

/*EXPORT OBJECT*/
if (typeof require == 'function'){
//require|exports
	exports.nas = nas
} else if (typeof navigator != 'undefined'){
//import|export
	export { nas };
}
